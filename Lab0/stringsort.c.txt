#include<stdio.h>
#include<string.h>
#include<stdlib.h>

int main() {
   char str[10][20]; 
   char temp[20];
   int i, j;

   for (i = 0; i < 10; i++) {
      printf("\nEnter the name %d: ", i);
     // flushall();
      gets(str[i]);
   }

   for (i = 0; i < 10; i++) {
      for (j = 0; j < 9; j++) {
         if (strcmp(str[j], str[j + 1]) > 0) {
            strcpy(temp, str[j]);
            strcpy(str[j], str[j + 1]);
            strcpy(str[j + 1], temp);
         }
      }
   }

  // flushall();

   printf("\nSorted List : ");
   for (i = 0; i < 9; i++){
      printf("%s, ",str[i]);
   }
   printf("%s\n",str[9]);
   return (0);
}
